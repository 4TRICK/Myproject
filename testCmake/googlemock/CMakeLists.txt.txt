cmake_minimum_required(VERSION 3.10)
project(MyProject)

# Включите подкаталоги с исходными кодами Google Test и Google Mock
add_subdirectory(testCmake/googletest)
add_subdirectory(testCmake/googlemock)

# Укажите, что вы хотите включить Google Test и Google Mock в ваш проект
include_directories(testCmake/googletest/include testCmake/googlemock/include)

# Ваш исполняемый файл и файлы исходного кода проекта
set(SOURCES
    main.cpp
    DataProcessor.cpp
    DataProcessorTest.cpp
)

# Ваш исполняемый файл
add_executable(MyProject ${SOURCES})

# Свяжите Google Test и Google Mock с вашим исполняемым файлом
target_link_libraries(MyProject gtest gtest_main gmock gmock_main)
